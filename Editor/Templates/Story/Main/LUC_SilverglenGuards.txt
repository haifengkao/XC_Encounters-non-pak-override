PROC
ProcCharInTriggerSpottedByChar((CHARACTER)_Player,(TRIGGER)_Trigger,(CHARACTER)_Guard)
AND
NOT DB_DefinedCompanion(_Player)
AND
DB_SilverglenGuardsIntro(_Trigger,_Guard)
THEN
ProcRemoveAllSilverGlenGuardTriggers();
ProcDoTwoSpeakerDialog("LUC_Guards_Intro",_Guard,_Player);

IF
DialogEnded("LUC_Guards_Intro",_ID)
AND
DB_DialogPlayers(_ID,_Player,1)
THEN
CharTimer(_Player,"LUC_SilverglenGuardsComment",3000);

IF
CharTimerFinished(_Player,"LUC_SilverglenGuardsComment")
THEN
Launch_PlayerComment(_Player,"LUC_SilverglenGuardsComment");

PROC
ProcRemoveAllSilverGlenGuardTriggers()
AND
DB_SilverglenGuardsIntro(_Trigger,_Guard)
THEN
NOT DB_SilverglenGuardsIntro(_Trigger,_Guard);
ProcTriggerUnregisterForPlayers(_Trigger);
